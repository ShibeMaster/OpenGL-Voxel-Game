#version 330 core

layout (location = 0) in vec3 pos;
layout (location = 1) in vec4 color;

out vec4 Color;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;
uniform vec3 test;


void main(){
	Color = color * vec4(test, 1.0);
	gl_Position = projection * view * model * vec4(pos, 1.0f);
}
